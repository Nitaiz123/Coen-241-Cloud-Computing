Vagrant.configure("2") do |config|
  config.vm.box = "ubuntu/bionic64"
  
  # Configure VirtualBox provider settings
  config.vm.provider "virtualbox" do |vb|
    vb.memory = 4048  # Set the VM's memory to 2048 MB
    vb.cpus = 4       # Assign 2 CPUs to the VM
  end
  
  # Set up synced folder between host and guest
  config.vm.synced_folder "qemu", "/", type: "virtualbox"
  
  # Provision the VM using a shell script
  config.vm.provision "shell", path: "vagrant_setup.sh"
end
